# Problema 14# Cree una variable en el conjunto de datos que indique la cantidad de vehículos que# tiene cada cliente y el valor total de sus vehículos. Denote estas nuevas variables# como "Vehículos" y "Valor_vehículos". Al nuevo conjunto de datos llámelo"Info_clientes_banco_ampliada"Vehículos = Info_clientes_banco$Vehículo_1 + Info_clientes_banco$Vehículo_2Valor_vehículos = Info_clientes_banco$Valor_vehículo_1 + Info_clientes_banco$Valor_vehículo_2Info_clientes_banco_ampliada = cbind(Info_clientes_banco,Vehículos,Valor_vehículos)# Problema 15# Obtenga un listado que contenga el valor total de los vehículos de los clientes# que tienen por lo menos un vehículo.Info_clientes_banco_ampliada[Info_clientes_banco_ampliada$Valor_vehículos!=0,]$Valor_vehículos# Problema 16# Genere un histograma con la información anterior# Histograma Frecuenciashist(Info_clientes_banco_ampliada[Info_clientes_banco_ampliada$Valor_vehículos!=0,]$Valor_vehículos,     main="Histograma Valor Vehículos", xlab="Valor Vehículos", ylab="Frecuencia")# Histograma porcentajeshist(Info_clientes_banco_ampliada[Info_clientes_banco_ampliada$Valor_vehículos!=0,]$Valor_vehículos,     freq=FALSE, main="Histograma Valor Vehículos", xlab="Valor Vehículos", ylab="Porcentaje")# Problema 17# Obtenga un listado de las identificaciones de los clientes que tienen cuenta# de ahorros o cuenta corrienteInfo_clientes_banco[Info_clientes_banco$Cuenta_corriente==1|Info_clientes_banco$Cuenta_ahorros==1,]$ID_cliente# Problema 18# Obtenga un listado de las identificaciones de los clientes que siendo propietarios# de dos inmuebles cuyo valor conjunto supere 600 millones, solicitaron crédito de vivienda.Info_clientes_banco$Valor_total_inmuebles = Info_clientes_banco$Valor_inmuble_1 + Info_clientes_banco$Valor_inmuble_2Info_clientes_banco[ Info_clientes_banco$Credito_vivienda==1 & Info_clientes_banco$Valor_total_inmuebles > 600000000,]$ID_cliente# Problema 19# Realice análisis descriptivos para los saldos en cuentas corrientes, de ahorros y# para los créditos de vivienda. Compare por sucursales, segmentos y tipos de cliente.# Interprete los resultados comparándolos. ¿Cuál de los asesores tiene mejor desempeño?# ¿Cuál sucursal? ¿Existen diferencias significativas entre los tipos de cliente?# ¿Tiene sentido la segmentación que hace el banco? Argumente sus respuestas con base# las medidas descriptivas y en gráficos apropiados.# :::::::::::::::: Medidas descriptivas cuentas corrientes VS sucursales ::::::::::::::::::::for( suc in c("AMERICAS", "CENTRO", "PRADO", "VILLA") ){  cat( "El promedio de la sucursal", suc, "es:", mean(Info_clientes_banco[(Info_clientes_banco$Cuenta_corriente==1 & Info_clientes_banco$Sucursal==suc),]$Saldo_cuenta_corriente), "\n" )  cat( "La mediana de la sucursal", suc, "es:", median(Info_clientes_banco[(Info_clientes_banco$Cuenta_corriente==1 & Info_clientes_banco$Sucursal==suc),]$Saldo_cuenta_corriente), "\n" )  cat( "Los cuartiles de la sucursal", suc, "son:", quantile(Info_clientes_banco[(Info_clientes_banco$Cuenta_corriente==1 & Info_clientes_banco$Sucursal==suc),]$Saldo_cuenta_corriente,probs = c(0.25,0.5,0.75)), "\n" )  cat( "La desviación estándar de la sucursal", suc, "es:", sd(Info_clientes_banco[(Info_clientes_banco$Cuenta_corriente==1 & Info_clientes_banco$Sucursal==suc),]$Saldo_cuenta_corriente), "\n" )  cat( "\n")}# Gráficos cuentas corrientes VS sucursalesboxplot( Info_clientes_banco[Info_clientes_banco$Cuenta_corriente==1,]$Saldo_cuenta_corriente ~ Info_clientes_banco[Info_clientes_banco$Cuenta_corriente==1,]$Sucursal,         main="Bloxplot Saldo Cuenta corrientes", xlab="Sucursal", ylab="Saldo", cex.axis = 0.7)# Gráfico cuentas corrientes VS asesoresboxplot( Info_clientes_banco[Info_clientes_banco$Cuenta_corriente==1,]$Saldo_cuenta_corriente ~ Info_clientes_banco[Info_clientes_banco$Cuenta_corriente==1,]$Asesor,          main="Bloxplot Saldo Cuenta corrientes", xlab="Asesor", ylab="Saldo", cex.axis = 0.5)# Gráfico cuentas corrientes VS tipo de clienteboxplot( Info_clientes_banco[Info_clientes_banco$Cuenta_corriente==1,]$Saldo_cuenta_corriente ~ Info_clientes_banco[Info_clientes_banco$Cuenta_corriente==1,]$Tipo_Cliente,          main="Bloxplot Saldo Cuenta corrientes", xlab="Cliente", ylab="Saldo", cex.axis = 0.4)# Gráfico cuentas corrientes VS segmentosboxplot( Info_clientes_banco[Info_clientes_banco$Cuenta_corriente==1,]$Saldo_cuenta_corriente ~ Info_clientes_banco[Info_clientes_banco$Cuenta_corriente==1,]$Segmento,          main="Bloxplot Saldo Cuenta corrientes", xlab="Segmento", ylab="Saldo", cex.axis = 0.4)
